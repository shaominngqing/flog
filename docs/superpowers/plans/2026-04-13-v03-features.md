/// SSE parsing without flog_net instrumentation.
static Stream<String> _passThroughSse(Stream<List<int>> byteStream) {
  final byteBuffer = <int>[];
  String lineBuffer = '';
  final controller = StreamController<String>();

  final subscription = byteStream.listen(
    (chunk) {
      byteBuffer.addAll(chunk);
      String decoded;
      try {
        decoded = utf8.decode(byteBuffer);
        byteBuffer.clear();
      } on FormatException {
        for (int i = 1; i <= 4 && i <= byteBuffer.length; i++) {
          try {
            decoded = utf8.decode(byteBuffer.sublist(0, byteBuffer.length - i));
            final remaining = byteBuffer.sublist(byteBuffer.length - i);
            byteBuffer
              ..clear()
              ..addAll(remaining);
            _processDecoded(decoded, lineBuffer, (newBuffer, data) {
              lineBuffer = newBuffer;
              if (data != null) controller.add(data);
            });
            return;
          } on FormatException {
            continue;
          }
        }
        return;
      }
      _processDecoded(decoded, lineBuffer, (newBuffer, data) {
        lineBuffer = newBuffer;
        if (data != null) controller.add(data);
      });
    },
    onError: (Object error) => controller.addError(error),
    onDone: () => controller.close(),
    cancelOnError: false,
  );

  controller.onCancel = () => subscription.cancel();
  return controller.stream;
}
```

import 'dart:async';
import 'dart:convert';

import 'package:dio/dio.dart';

import 'flog_http_interceptor.dart';
import 'flog_net.dart' show flogEnabled;
import 'flog_sse_parser.dart';

/// Configuration for the flog HTTP interceptor inside [FlogDio].
class FlogHttpConfig {
  final bool includeRequestHeaders;
  final bool includeResponseHeaders;
  final bool includeRequestBody;
  final bool includeResponseBody;
  final int maxBodySize;
  final FlogHttpFilter? filter;

  const FlogHttpConfig({
    this.includeRequestHeaders = true,
    this.includeResponseHeaders = true,
    this.includeRequestBody = true,
    this.includeResponseBody = true,
    this.maxBodySize = 10 * 1024 * 1024,
    this.filter,
  });
}

/// Response from an SSE endpoint.
class SseResponse {
  /// Response headers from the server.
  final Headers headers;

  /// HTTP status code.
  final int? statusCode;

  /// Stream of parsed SSE data payloads (with `data:` prefix and `[DONE]`
  /// already stripped).
  final Stream<String> stream;

  SseResponse({
    required this.headers,
    required this.statusCode,
    required this.stream,
  });
}

/// Drop-in [Dio] replacement with automatic flog instrumentation.
///
/// ```dart
/// final dio = FlogDio(baseUrl: 'https://api.example.com');
/// final response = await dio.get('/users');
///
/// final sse = await dio.sse('/chat/stream', method: 'POST', data: body);
/// await for (final event in sse.stream) { ... }
/// ```
class FlogDio extends Dio {
  FlogDio({
    String? baseUrl,
    FlogHttpConfig? flogConfig,
    BaseOptions? options,
  }) : super(options ?? BaseOptions(baseUrl: baseUrl ?? '')) {
    if (baseUrl != null && options == null) {
      this.options.baseUrl = baseUrl;
    }
    if (flogEnabled) {
      final config = flogConfig ?? const FlogHttpConfig();
      interceptors.insert(
        0,
        FlogHttpInterceptor(
          includeRequestHeaders: config.includeRequestHeaders,
          includeResponseHeaders: config.includeResponseHeaders,
          includeRequestBody: config.includeRequestBody,
          includeResponseBody: config.includeResponseBody,
          maxBodySize: config.maxBodySize,
          filter: config.filter,
        ),
      );
    }
  }

  /// Send an SSE request and return a parsed event stream.
  ///
  /// Internally sets `ResponseType.stream`, parses SSE frames, and wraps
  /// with flog_net instrumentation (when enabled).
  Future<SseResponse> sse(
    String path, {
    String method = 'GET',
    dynamic data,
    Options? options,
    Map<String, dynamic>? queryParameters,
  }) async {
    final opts = (options ?? Options()).copyWith(
      method: method,
      responseType: ResponseType.stream,
    );

    final response = await request<ResponseBody>(
      path,
      data: data,
      options: opts,
      queryParameters: queryParameters,
    );

    final rawStream = response.data!.stream;
    final fullUrl = response.realUri.toString();

    final Stream<String> parsed;
    if (flogEnabled) {
      parsed = FlogSseParser.wrap(rawStream, url: fullUrl, method: method);
    } else {
      parsed = FlogSseParser.wrap(rawStream, url: fullUrl, method: method);
    }

    return SseResponse(
      headers: response.headers,
      statusCode: response.statusCode,
      stream: parsed,
    );
  }
}
```

```rust
//! Replay HTTP requests from captured NetworkEntry data.

use std::sync::Arc;
use tokio::sync::Mutex;

use crate::app::App;
use crate::domain::network::{EntrySource, NetworkEntry, NetworkStatus, Protocol};

/// Replay the given network entry asynchronously.
/// Inserts result as a new entry with `EntrySource::Replay`.
pub async fn replay_request(app: Arc<Mutex<App>>, entry: NetworkEntry) {
    let client = reqwest::Client::new();

    let method = match entry.method.to_uppercase().as_str() {
        "GET" => reqwest::Method::GET,
        "POST" => reqwest::Method::POST,
        "PUT" => reqwest::Method::PUT,
        "DELETE" => reqwest::Method::DELETE,
        "PATCH" => reqwest::Method::PATCH,
        "HEAD" => reqwest::Method::HEAD,
        "OPTIONS" => reqwest::Method::OPTIONS,
        _ => reqwest::Method::GET,
    };

    let mut builder = client.request(method.clone(), &entry.url);

    // Restore headers from original request
    if let Some(ref headers_json) = entry.request_headers {
        if let Ok(serde_json::Value::Object(map)) = serde_json::from_str(headers_json) {
            for (key, val) in &map {
                let val_str = match val {
                    serde_json::Value::String(s) => s.clone(),
                    serde_json::Value::Array(arr) => {
                        arr.iter()
                            .filter_map(|v| v.as_str())
                            .collect::<Vec<_>>()
                            .join(", ")
                    }
                    other => other.to_string(),
                };
                builder = builder.header(key.as_str(), val_str);
            }
        }
    }

    // Restore body from original request
    if let Some(ref body) = entry.request_body {
        if !body.is_empty() {
            builder = builder.body(body.clone());
        }
    }

    let start = std::time::Instant::now();

    {
        let mut a = app.lock().await;
        a.show_status("Replaying...".to_string());
    }

    let result = builder.send().await;
    let duration = start.elapsed().as_millis() as u64;

    let mut a = app.lock().await;
    let id = a.network_store.len() as u64 + 10_000_000; // high ID to avoid collision

    let mut replay_entry = NetworkEntry::new_http(
        id,
        method.to_string(),
        entry.url.clone(),
        String::new(),
    );
    replay_entry.source = EntrySource::Replay;
    replay_entry.request_headers = entry.request_headers.clone();
    replay_entry.request_body = entry.request_body.clone();

    match result {
        Ok(resp) => {
            let status = resp.status().as_u16();
            let headers = format!("{:?}", resp.headers());
            let body = resp.text().await.unwrap_or_default();
            let body_size = body.len() as u64;

            replay_entry.status = NetworkStatus::Completed;
            replay_entry.http_status = Some(status);
            replay_entry.duration = Some(duration);
            replay_entry.response_headers = Some(headers);
            replay_entry.response_size = Some(body_size);
            replay_entry.response_body = Some(body);

            a.show_status(format!("Replay: {} ({duration}ms)", status));
        }
        Err(err) => {
            replay_entry.status = NetworkStatus::Failed;
            replay_entry.duration = Some(duration);
            replay_entry.error = Some(err.to_string());

            a.show_status(format!("Replay failed: {}", err));
        }
    }

    a.network_store.push_entry(replay_entry);
    a.network.mark_filter_dirty();
}
```

            entry.1 += d; // total duration
        }
        if e.status == NetworkStatus::Failed {
            entry.2 += 1; // errors
        }

    domain_vec.sort_by(|a, b| b.1.0.cmp(&a.1.0));

    let domain_header = Row::new(vec![
        Cell::from("Domain").style(Style::default().fg(SUBTEXT0).add_modifier(Modifier::BOLD)),
        Cell::from("Requests").style(Style::default().fg(SUBTEXT0).add_modifier(Modifier::BOLD)),
        Cell::from("Avg Duration").style(Style::default().fg(SUBTEXT0).add_modifier(Modifier::BOLD)),
        Cell::from("Error Rate").style(Style::default().fg(SUBTEXT0).add_modifier(Modifier::BOLD)),

    let domain_rows: Vec<Row> = domain_vec.iter().take(max_domain_rows).map(|(domain, (count, total_dur, errors))| {
        let avg_dur = if *count > 0 { total_dur / *count as u64 } else { 0 };
        let err_rate = if *count > 0 { (*errors as f64 / *count as f64) * 100.0 } else { 0.0 };
        Row::new(vec![
            Cell::from(domain.clone()).style(Style::default().fg(TEAL)),
            Cell::from(format!("{}", count)).style(Style::default().fg(TEXT)),
            Cell::from(format_duration(avg_dur)).style(Style::default().fg(duration_color(avg_dur))),
            Cell::from(format!("{:.1}%", err_rate)).style(

                else if err_rate > 0.0 { Style::default().fg(YELLOW) }
                else { Style::default().fg(GREEN) }

fn percentile(sorted: &[u64], p: usize) -> u64 {
    if sorted.is_empty() {
        return 0;
    }
    let idx = (p as f64 / 100.0 * (sorted.len() - 1) as f64).round() as usize;
    sorted[idx.min(sorted.len() - 1)]
}

fn format_duration(ms: u64) -> String {
    if ms >= 1000 {
        format!("{:.1}s", ms as f64 / 1000.0)
    } else {
        format!("{}ms", ms)
    }
}

fn duration_color(ms: u64) -> Color {
    if ms > 1000 { RED } else if ms > 500 { YELLOW } else { GREEN }
}

fn truncate_url(s: &str, max: usize) -> String {
    if s.len() > max {
        format!("{}...", &s[..max - 3])
    } else {
        s.to_string()
    }
}

fn status_row(label: &str, count: u32, total: f64, color: Color) -> Row<'static> {
    let pct = (count as f64 / total) * 100.0;
    Row::new(vec![
        Cell::from(label.to_string()).style(Style::default().fg(color)),
        Cell::from(format!("{}", count)).style(Style::default().fg(TEXT)),
        Cell::from(format!("{:.1}%", pct)).style(Style::default().fg(SUBTEXT0)),
    ])
}

fn extract_domain(url: &str) -> String {
    if let Some(pos) = url.find("://") {
        let after_scheme = &url[pos + 3..];
        if let Some(slash) = after_scheme.find('/') {
            return after_scheme[..slash].to_string();
        }
        return after_scheme.to_string();
    }
    url.to_string()
}

//! Mock rule matching engine.

#[derive(Debug, Clone)]
pub struct MockRule {
    pub id: usize,
    pub url_pattern: String,
    pub method: Option<String>,
    pub status_code: u16,
    pub response_body: String,
    pub delay_ms: u64,
    pub enabled: bool,
    pub hit_count: u32,
}

impl MockRule {
    pub fn matches(&self, url: &str, method: &str) -> bool {
        if !self.enabled {
            return false;
        }
        if let Some(ref m) = self.method {
            if !m.eq_ignore_ascii_case(method) {
                return false;
            }
        }
        url.contains(&self.url_pattern)
    }
}

#[derive(Debug, Default)]
pub struct MockRuleStore {
    rules: Vec<MockRule>,
    next_id: usize,
}

impl MockRuleStore {
    pub fn new() -> Self {
        Self { rules: Vec::new(), next_id: 1 }
    }

    pub fn add(&mut self, url_pattern: String, method: Option<String>, status_code: u16, response_body: String, delay_ms: u64) -> usize {
        let id = self.next_id;
        self.next_id += 1;
        self.rules.push(MockRule {
            id, url_pattern, method, status_code, response_body, delay_ms,
            enabled: true, hit_count: 0,
        });
        id
    }

    pub fn find_match(&mut self, url: &str, method: &str) -> Option<&MockRule> {
        // Find matching rule, increment hit count
        for rule in &mut self.rules {
            if rule.matches(url, method) {
                rule.hit_count += 1;
                return Some(rule);
            }
        }
        None
    }

    pub fn rules(&self) -> &[MockRule] {
        &self.rules
    }

    pub fn toggle(&mut self, id: usize) {
        if let Some(r) = self.rules.iter_mut().find(|r| r.id == id) {
            r.enabled = !r.enabled;
        }
    }

    pub fn remove(&mut self, id: usize) {
        self.rules.retain(|r| r.id != id);
    }

    pub fn get_mut(&mut self, id: usize) -> Option<&mut MockRule> {
        self.rules.iter_mut().find(|r| r.id == id)
    }

    pub fn len(&self) -> usize {
        self.rules.len()
    }

    pub fn enabled_count(&self) -> usize {
        self.rules.iter().filter(|r| r.enabled).count()
    }
}

//! Local HTTP proxy server for Mock and traffic inspection.

pub mod mock;

use std::sync::Arc;
use tokio::sync::Mutex;

use bytes::Bytes;
use http_body_util::{BodyExt, Full};
use hyper::server::conn::http1;
use hyper::service::service_fn;
use hyper::{Request, Response, StatusCode};
use hyper_util::rt::TokioIo;
use tokio::net::TcpListener;

use crate::app::App;
use crate::domain::network::{EntrySource, NetworkEntry, NetworkStatus};

const START_PORT: u16 = 9999;
const MAX_ATTEMPTS: u16 = 10;

/// Start the proxy server. Tries ports 9999..10008.
/// Returns the actual bound port.
pub async fn start_proxy(app: Arc<Mutex<App>>) -> Result<u16, String> {
    let mut port = START_PORT;
    let listener = loop {
        match TcpListener::bind(format!("127.0.0.1:{}", port)).await {
            Ok(l) => break l,
            Err(_) if port < START_PORT + MAX_ATTEMPTS => {
                port += 1;
            }
            Err(e) => return Err(format!("Could not bind proxy: {}", e)),
        }
    };

    {
        let mut a = app.lock().await;
        a.proxy_port = Some(port);
        a.proxy_running = true;
    }

    let app_clone = Arc::clone(&app);
    tokio::spawn(async move {
        loop {
            let (stream, _) = match listener.accept().await {
                Ok(s) => s,
                Err(_) => continue,
            };

            let app_req = Arc::clone(&app_clone);
            tokio::spawn(async move {
                let io = TokioIo::new(stream);
                let svc = service_fn(move |req| {
                    let app_inner = Arc::clone(&app_req);
                    async move { handle_request(app_inner, req).await }
                });
                let _ = http1::Builder::new().serve_connection(io, svc).await;
            });
        }
    });

    Ok(port)
}

async fn handle_request(
    app: Arc<Mutex<App>>,
    req: Request<hyper::body::Incoming>,
) -> Result<Response<Full<Bytes>>, hyper::Error> {
    let method = req.method().to_string();
    let uri = req.uri().to_string();

    // Check mock rules
    let mock_match = {
        let mut a = app.lock().await;
        a.mock_rules.find_match(&uri, &method).map(|r| {
            (r.status_code, r.response_body.clone(), r.delay_ms)
        })
    };

    if let Some((status_code, body, delay_ms)) = mock_match {
        // Delay if configured
        if delay_ms > 0 {
            tokio::time::sleep(std::time::Duration::from_millis(delay_ms)).await;
        }

        // Record as mocked entry
        let mut a = app.lock().await;
        let id = a.network_store.len() as u64 + 20_000_000;
        let mut entry = NetworkEntry::new_http(id, method, uri, String::new());
        entry.source = EntrySource::Mocked;
        entry.status = NetworkStatus::Completed;
        entry.http_status = Some(status_code);
        entry.response_body = Some(body.clone());
        entry.duration = Some(delay_ms);
        a.network_store.push_entry(entry);
        a.network.mark_filter_dirty();

        let status = StatusCode::from_u16(status_code).unwrap_or(StatusCode::OK);
        return Ok(Response::builder()
            .status(status)
            .body(Full::new(Bytes::from(body)))
            .unwrap());
    }

    // No mock match — forward to real server
    let client = reqwest::Client::new();
    let req_method = match method.as_str() {
        "GET" => reqwest::Method::GET,
        "POST" => reqwest::Method::POST,
        "PUT" => reqwest::Method::PUT,
        "DELETE" => reqwest::Method::DELETE,
        "PATCH" => reqwest::Method::PATCH,
        "HEAD" => reqwest::Method::HEAD,
        "OPTIONS" => reqwest::Method::OPTIONS,
        _ => reqwest::Method::GET,
    };

    // Collect request body
    let body_bytes = req.collect().await.map(|b| b.to_bytes()).unwrap_or_default();
    let body_str = String::from_utf8_lossy(&body_bytes).to_string();

    let start = std::time::Instant::now();
    let result = client.request(req_method, &uri)
        .body(body_bytes.clone())
        .send()
        .await;
    let duration = start.elapsed().as_millis() as u64;

    match result {
        Ok(resp) => {
            let status = resp.status().as_u16();
            let resp_body = resp.text().await.unwrap_or_default();

            Ok(Response::builder()
                .status(StatusCode::from_u16(status).unwrap_or(StatusCode::OK))
                .body(Full::new(Bytes::from(resp_body)))
                .unwrap())
        }
        Err(e) => {
            Ok(Response::builder()
                .status(StatusCode::BAD_GATEWAY)
                .body(Full::new(Bytes::from(format!("Proxy error: {}", e))))
                .unwrap())
        }
    }
}
